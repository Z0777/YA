function parseXL(buf){
  const wb=XLSX.read(buf,{type:'array',cellDates:true});const ws=wb.Sheets[wb.SheetNames[0]];
  const json=XLSX.utils.sheet_to_json(ws,{header:1,defval:''});if(json.length<2)return;
  const hdr=json[0].map(h=>String(h).trim());const rows=json.slice(1).filter(r=>r.some(c=>c!==''));
  const mp=dCols(hdr);

  PD=rows.map(row=>{
    // Hours: from Hora Inicio / Hora Fin (DateTime fields)
    const hi=mp.hora_inicio!==-1?row[mp.hora_inicio]:null;
    const hf=mp.hora_fin!==-1?row[mp.hora_fin]:null;
    let hrs=0;
    if(hi instanceof Date&&hf instanceof Date){hrs=Math.max(0,(hf-hi)/3600000);}
    else{hrs=cH(hi,hf);}
    if(hrs>16)hrs=0;

    const plots=parseFloat(mp.plots!==-1?row[mp.plots]:0)||0;
    // Fecha from Fecha Inicio or from Hora Inicio date part
    let fecha='';
    const rawF=mp.fecha_inicio!==-1?row[mp.fecha_inicio]:(hi instanceof Date?hi:null);
    if(rawF instanceof Date)fecha=rawF.toISOString().split('T')[0];
    else if(rawF){const d=new Date(rawF);if(!isNaN(d))fecha=d.toISOString().split('T')[0];}

    return{fecha,nombre:g(row,mp.nombre),set:g(row,mp.set),mtc:g(row,mp.mtc),cultivo:g(row,mp.cultivo),
      actividad:g(row,mp.actividad),sub:g(row,mp.subactividad),plots,hours:hrs,rate:hrs>0?plots/hrs:0};
  }).filter(d=>d.nombre&&d.plots>0);

  document.getElementById('loader').classList.add('hide');
  document.getElementById('app').style.display='flex';
  buildAll();
  const now=new Date();document.getElementById('sInfo').textContent=`${PD.length} reg · ${now.toLocaleTimeString('es-MX',{hour:'2-digit',minute:'2-digit'})}`;
}

function g(row,idx){return idx!==-1?String(row[idx]||'').trim():'';}
function dCols(hdr){
  const mp={};const hl=hdr.map(h=>h.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g,''));
  Object.keys(HN).forEach(key=>{let best=-1;
    for(let i=0;i<hl.length;i++){for(const hint of HN[key]){const hn=hint.normalize('NFD').replace(/[\u0300-\u036f]/g,'');if(hl[i]===hn||hl[i].includes(hn)){best=i;break;}}if(best!==-1)break;}
    mp[key]=best;});return mp;
}
function cH(s,e){
  if(!s||!e)return 0;const pt=v=>{if(v instanceof Date)return v.getHours()+v.getMinutes()/60;
    const str=String(v).trim();const n=parseFloat(str);if(!isNaN(n)&&n>=0&&n<=1)return n*24;
    const m=str.match(/(\d{1,2})[:\.](\d{2})/);if(m)return parseInt(m[1])+parseInt(m[2])/60;return NaN;};
  const sv=pt(s),ev=pt(e);if(isNaN(sv)||isNaN(ev))return 0;let d=ev-sv;if(d<0)d+=24;return d;
}

async function exportSection(id){
  const el=document.getElementById(id);
  const{jsPDF}=window.jspdf;
  const canvas=await html2canvas(el,{scale:2,useCORS:true,backgroundColor:'#FFFFFF'});
  const imgData=canvas.toDataURL('image/png');
  const pdf=new jsPDF({orientation:'landscape',unit:'mm',format:'a4'});
  const pW=pdf.internal.pageSize.getWidth()-20;
  const pH=(canvas.height*pW)/canvas.width;
  pdf.addImage(imgData,'PNG',10,10,pW,Math.min(pH,pdf.internal.pageSize.getHeight()-20));
  pdf.save(`SPA_${id}_${new Date().toISOString().split('T')[0]}.pdf`);
}
